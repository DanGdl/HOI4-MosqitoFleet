	    #    cancel_if_not_visible = yes

	    custom_cost_text = mdg_decision_cost_12_convoy_3_NavXP
		custom_cost_trigger = {
			has_navy_experience > 2
			has_equipment = {
				convoy > 12
			}
		}

		available = {	
			has_equipment = {
   				convoy > 12
	  		}	
	  		if = {
				limit = {
					has_dlc = "Man the Guns"
				}
				has_tech = early_ship_hull_light
			}
			else = {
				has_tech = early_destroyer
			}
		}
		days_remove = 30

		complete_effect = {
			add_equipment_to_stockpile = { 
				type = convoy_1
				amount = -12
			}
		}
		remove_effect = {
			if = {
				limit = {
					has_dlc = "Man the Guns"
				}
				if = {
					limit = {
						NOT = {	has_country_flag = mdg_flag_refit_to_mine_layer_mtg }
					}
					set_country_flag = mdg_flag_refit_to_mine_layer_mtg
					
					create_equipment_variant = {	
						name = "Refitted Mine Layer" 		
						type = mdg_ship_hull_auxiliary_1
						name_group = "DEN_DD_HISTORICAL" # Note: Doesn't matter if ROOT is not DEN, the effect will grab the appropriate country-specific name list anyways
						parent_version = 0
						modules = {
							fixed_ship_battery_slot = ship_light_battery_1
							fixed_ship_anti_air_slot = empty
							fixed_ship_fire_control_system_slot = empty
							fixed_ship_engine_slot = light_ship_engine_1
							fixed_ship_secondaries_slot = empty
							mid_1_custom_slot = empty
							mid_2_custom_slot = empty
							rear_1_custom_slot = ship_mine_layer_1
							rear_2_custom_slot = ship_mine_layer_1
						}
						obsolete = no
					}
				}
				create_ship = {
					type = mdg_ship_hull_auxiliary_1
					equipment_variant = "Refitted Mine Layer"
					creator = ROOT
					name = "AUX"
				}
			}
			# NO MTG
			else = {
				if = {
					limit = {
						NOT = {	has_country_flag = mdg_flag_refit_to_mine_layer_vanila }
					}
					set_country_flag = mdg_flag_refit_to_mine_layer_vanila

					create_equipment_variant = {
						name = "Refitted Mine Layer"
						type = mdg_auxiliary_1
						upgrades = {
							ship_reliability_upgrade = 0
							ship_engine_upgrade = 0
							ship_gun_upgrade = 0
							ship_anti_air_upgrade = 0
						}
						obsolete = no
					}
				}
				create_ship = {
					type = mdg_auxiliary_1
					equipment_variant = "Refitted Mine Layer"
					creator = ROOT
					name = "AUX"
				}		
			}
		}
